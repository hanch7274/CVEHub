이벤트 흐름 시뮬레이션

1. React 컴포넌트에서 CVE 구독 요청
2. useSubscription 훅이 webSocketInstance.subscribe() 호출
3. WebSocketService가 EventSystem에 구독 등록
4. WebSocketService가 subscribeToCVE()를 통해 서버에 구독 메시지 전송
5. 서버로부터 구독 확인 메시지 수신
6. MessageManager가 메시지 변환 및 처리
7. EventSystem이 subscription:{cveId} 이벤트 발생
8. useSubscription 훅의 콜백 함수 실행
9. React 컴포넌트 상태 업데이트 및 UI 갱신

코드 예시

// 컴포넌트에서 사용 예시
const MyComponent = ({ cveId }) => {
  const [subscribers, setSubscribers] = useState([]);
  
  // useSubscription 훅 사용
  const { refreshSubscription } = useSubscription(
    cveId,
    (data) => console.log('CVE 업데이트:', data),
    (newSubscribers) => setSubscribers(newSubscribers)
  );
  
  return (
    <div>
      <h3>현재 구독자: {subscribers.length}명</h3>
      <button onClick={refreshSubscription}>새로고침</button>
    </div>
  );
};
